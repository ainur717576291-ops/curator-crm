  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,

  status TEXT DEFAULT 'active',
  problem TEXT,
  progress TEXT,
  results TEXT,
  difficulties TEXT,
  notes TEXT,
  modules JSONB DEFAULT '[]',
  created_at TIMESTAMPTZ DEFAULT NOW()
